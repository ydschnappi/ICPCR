367 class BitField16 {
368 public:
369 BitField16() : value_(0) {}
370
371 // Sets the value of the specified group to the given value
372 void setGroupValue(uint16_t group_index, uint16_t group_value) {
373 assert(group_value < 16 && "group_value must be less than 16");
374 assert(group_index < 4 && "group_index must be less than 4");
375
376 // calculate the bit position of the start of the group you want to set
377 const uint16_t startBit = group_index * group_size_;
378 const auto setMask = static_cast<uint16_t>(group_value << startBit);
379
380 // clear the bits in the group
381 value_ &= ~(0xF << startBit);
382
383 // set the bits in the group
384 value_ |= setMask;
385 }
386
387 [[nodiscard]] uint16_t getGroup(uint16_t group_index) const {
388 assert(group_index < 4 && "group_index must be less than 4");
389 uint16_t startBit = group_index * group_size_;
390 return (value_ >> startBit) & 0xF;
391 }
392
393 void clear() { value_ = 0; }
394 [[nodiscard]] uint16_t get() const { return value_; }
395
396 private:
397 static constexpr uint16_t group_size_ = 4; // size of each group
398 uint16_t value_ = 0;
399 };

646 [[nodiscard]] inline std::vector<std::string_view> splitString(std::string_view string,
647 const char& delimiter) {
648 std::vector<std::string_view> result;
649 size_t start = 0;
650 size_t end = string.find(delimiter);
651
652 while (end != std::string_view::npos) {
653 result.push_back(string.substr(start, end - start));
654 start = end + 1;
655 end = string.find(delimiter, start);
656 }
657
658 // Add the last chunk (or the only chunk if there are no delimiters)
659 result.push_back(string.substr(start));
660
661 return result;
662 }

1298 const std::function<Bitboard(Square, Bitboard)>& attacks) {
1299 const Bitboard edges =
1300 ((MASK_RANK[static_cast<int>(Rank::RANK_1)] | MASK_RANK[static_cast<int>(Rank::RANK_8)]) &
1301 ~MASK_RANK[static_cast<int>(utils::squareRank(sq))]) |
1302 ((MASK_FILE[static_cast<int>(File::FILE_A)] | MASK_FILE[static_cast<int>(File::FILE_H)]) &
1303 ~MASK_FILE[static_cast<int>(utils::squareFile(sq))]);

1305 Bitboard occ = 0ULL;
1306
1307 table[sq].magic = magic;
1308 table[sq].mask = attacks(sq, occ) & ~edges;
1309 table[sq].shift = constants::MAX_SQ - builtin::popcount(table[sq].mask);
1310
1311 if (sq < constants::MAX_SQ - 1) {
1312 table[sq + 1].attacks = table[sq].attacks + (1ULL<< builtin::popcount(table[sq].mask));
1313 }
1314
1315 do {
1316 table[sq].attacks[table[sq](occ)] = attacks(sq, occ);
1317 occ = (occ - table[sq].mask) & table[sq].mask;
1318 } while (occ);
1319 }

1553 inline void Board::setFenInternal(std::string_view fen) {

1557

1559 while (fen[0] == ' ') fen.remove_prefix(1);
1560
1561 occ_all_ = 0ULL;
1562
1563 for (const auto c : { Color::WHITE, Color::BLACK }) {
1564 for (int i = 0; i < 6; i++) {
1565 pieces_bb_[static_cast<int>(c)][i] = 0ULL;
1566 }
1567 }
1568
1569 const auto params = utils::splitString(fen, ' ');
1570
1571 const auto position = params[0];
1572 const auto move_right = params[1];
1573 const auto castling = params[2];
1574 const auto en_passant = params[3];
1575
1576 if (params.size() > 4) {
1577 std::from_chars(params[4].data(), params[4].data() + params[4].size(), half_moves_);
1578 }
1579 else {
1580 half_moves_ = 0;
1581 }
1582
1583 if (params.size() > 5) {
1584 std::from_chars(params[5].data(), params[5].data() + params[5].size(), plies_played_);
1585 plies_played_ = plies_played_ * 2 - 2;
1586 }
1587 else {
1588 plies_played_ = 0;
1589 }
1590
1591 side_to_move_ = (move_right == "w") ? Color::WHITE : Color::BLACK;
1592
1593 if (side_to_move_ == Color::BLACK) {
1594 plies_played_++;
1595 }

1597 auto square = Square(56);
1598 for (char curr : position) {
1599 if (charToPiece(curr) != Piece::NONE) {
1600 const Piece piece = charToPiece(curr);
1601 placePiece(piece, square);
1602 hash_key_ ^= zobrist::piece(piece, square);
1603
1604 square = Square(square + 1);
1605 }
1606 else if (curr == '/')
1607 square = Square(square - 16);
1608 else if (isdigit(curr)) {
1609 square = Square(square + (curr - '0'));
1610 }
1611 }
1612
1613 castling_rights_.clearAllCastlingRights();
1614
1615 for (char i : castling) {
1616 if (i == '-') break;
1617 if (!chess960_) {
1618 if (i == 'K')
1619 castling_rights_
1620 .setCastlingRight<Color::WHITE, CastleSide::KING_SIDE, File::FILE_H>();
1621 if (i == 'Q')
1622 castling_rights_
1623 .setCastlingRight<Color::WHITE, CastleSide::QUEEN_SIDE, File::FILE_A>();
1624 if (i == 'k')
1625 castling_rights_

1627 if (i == 'q')
1628 castling_rights_
1629 .setCastlingRight<Color::BLACK, CastleSide::QUEEN_SIDE, File::FILE_A>();
1630 }
1631 else {
1632 if (i == 'K' || i == 'k') {
1633 // find rook on the right side of the king
1634 const auto color = isupper(i) ? Color::WHITE : Color::BLACK;
1635 const auto king_sq = kingSq(color);
1636
1637 const auto sq_corner = color == Color::WHITE ? SQ_H1 : SQ_H8;
1638
1639 for (int sq = king_sq + 1; sq <= sq_corner; sq++) {
1640 if (at<PieceType>(Square(sq)) == PieceType::NONE) continue;
1641 if (at<PieceType>(Square(sq)) == PieceType::ROOK &&
1642 int(at(Square(sq))) / 6 == int(color)) {
1643 castling_rights_.setCastlingRight(color, CastleSide::KING_SIDE,
1644 utils::squareFile(Square(sq)));
1645 break;
1646 }
1647 }
1648
1649 }
1650 else if (i == 'Q' || i == 'q') {
1651 // find rook on the left side of the king
1652 const auto color = isupper(i) ? Color::WHITE : Color::BLACK;
1653 const auto king_sq = kingSq(color);
1654
1655 const auto sq_corner = color == Color::WHITE ? SQ_A1 : SQ_A8;
1656
1657 for (int sq = king_sq - 1; sq >= sq_corner; sq--) {
1658 if (at<PieceType>(Square(sq)) == PieceType::NONE) continue;
1659 if (at<PieceType>(Square(sq)) == PieceType::ROOK &&
1660 int(at(Square(sq))) / 6 == int(color)) {
1661 castling_rights_.setCastlingRight(color, CastleSide::QUEEN_SIDE,
1662 utils::squareFile(Square(sq)));
1663 break;
1664 }
1665 }
1666 }
1667 else {
1668 const auto color = isupper(i) ? Color::WHITE : Color::BLACK;
1669 const auto king_sq = kingSq(color);
1670 const auto file = static_cast<File>(tolower(i) - 97);
1671 const auto side = int(file) > int(utils::squareFile(king_sq))
1672 ? CastleSide::KING_SIDE
1673 : CastleSide::QUEEN_SIDE;
1674 castling_rights_.setCastlingRight(color, side, file);
1675 }
1676 }
1677 }
1678
1679 if (en_passant == "-") {
1680 enpassant_sq_ = NO_SQ;
1681 }
1682 else {
1683 const char letter = en_passant[0];
1684 const int file = letter - 96;
1685 const int rank = en_passant[1] - 48;
1686 enpassant_sq_ = Square((rank - 1) * 8 + file - 1);
1687 }
1688
1689 hash_key_ = zobrist();
1690 occ_all_ = all();
1691
1692 prev_states_.clear();
1693 prev_states_.reserve(150);
1694 }

1698 [[nodiscard]] inline std::string Board::getFen(bool move_counters) const {
1699 std::stringstream ss;
1700
1701 // Loop through the ranks of the board in reverse order
1702 for (int rank = 7; rank >= 0; rank--) {
1703 int free_space = 0;
1704
1705 // Loop through the files of the board
1706 for (int file = 0; file < 8; file++) {
1707 // Calculate the square index
1708 const int sq = rank * 8 + file;
1709
1710 // Get the piece at the current square
1711 const Piece piece = at(Square(sq));
1712
1713 // If there is a piece at the current square
1714 if (piece != Piece::NONE) {
1715 // If there were any empty squares before this piece,
1716 // append the number of empty squares to the FEN string
1717 if (free_space) {
1718 ss << free_space;
1719 free_space = 0;
1720 }
1721
1722 // Append the character representing the piece to the FEN string
1723 ss << pieceToChar(piece);
1724 }
1725 else {
1726 // If there is no piece at the current square, increment the
1727 // counter for the number of empty squares
1728 free_space++;
1729 }
1730 }
1731
1732 // If there are any empty squares at the end of the rank,
1733 // append the number of empty squares to the FEN string
1734 if (free_space != 0) {
1735 ss << free_space;
1736 }
1737
1738 // Append a "/" character to the FEN string, unless this is the last rank
1739 ss << (rank > 0 ? "/" : "");
1740 }
1741
1742 // Append " w " or " b " to the FEN string, depending on which player's turn it is
1743 ss << (side_to_move_ == Color::WHITE ? " w " : " b ");
1744
1745 // Append the appropriate characters to the FEN string to indicate
1746 // whether castling is allowed for each player
1747 ss << getCastleString();
1748 if (castling_rights_.isEmpty()) ss << "-";
1749
1750 // Append information about the en passant square (if any)
1751 // and the half-move clock and full move number to the FEN string
1752 if (enpassant_sq_ == NO_SQ)
1753 ss << " - ";
1754 else
1755 ss << " " << squareToString[enpassant_sq_] << " ";
1756
1757 if (move_counters) ss << halfMoveClock() << " " << fullMoveNumber();
1758
1759 // Return the resulting FEN string
1760 return ss.str();
1761 }

1763 [[nodiscard]] inline U64 Board::zobrist() const {
1764 U64 hash_key = 0ULL;
1765
1766 U64 wPieces = us(Color::WHITE);
1767 U64 bPieces = us(Color::BLACK);
1768
1769 while (wPieces) {
1770 const Square sq = builtin::poplsb(wPieces);
1771 hash_key ^= zobrist::piece(at(sq), sq);
1772 }
1773 while (bPieces) {
1774 const Square sq = builtin::poplsb(bPieces);
1775 hash_key ^= zobrist::piece(at(sq), sq);
1776 }
1777
1778 U64 ep_hash = 0ULL;
1779 if (enpassant_sq_ != NO_SQ) ep_hash ^= zobrist::enpassant(utils::squareFile(enpassant_sq_));
1780
1781 U64 side_to_move_hash = 0ULL;
1782 if (side_to_move_ == Color::WHITE) side_to_move_hash ^= zobrist::sideToMove();
1783
1784 // Castle hash
1785

1787 castling_hash ^= zobrist::castling(castling_rights_.getHashIndex());
1788
1789 return hash_key ^ ep_hash ^ side_to_move_hash ^ castling_hash;
1790 }

1937 inline void Board::makeMove(const Move& move) {
1938 const auto capture = at(move.to()) != Piece::NONE && move.typeOf() != Move::CASTLING;
1939 const auto captured = at(move.to());
1940 const auto pt = at<PieceType>(move.from());
1941
1942 prev_states_.emplace_back(hash_key_, castling_rights_, enpassant_sq_, half_moves_, captured);
1943
1944 half_moves_++;
1945 plies_played_++;
1946
1947 if (enpassant_sq_ != NO_SQ) hash_key_ ^= zobrist::enpassant(utils::squareFile(enpassant_sq_));
1948 enpassant_sq_ = NO_SQ;
1949
1950 if (capture) {
1951 half_moves_ = 0;
1952
1953 hash_key_ ^= zobrist::piece(captured, move.to());
1954 removePiece(captured, move.to());
1955
1956 const auto rank = utils::squareRank(move.to());
1957
1958 if (utils::typeOfPiece(captured) == PieceType::ROOK &&
1959 ((rank == Rank::RANK_1 && side_to_move_ == Color::BLACK) ||
1960 (rank == Rank::RANK_8 && side_to_move_ == Color::WHITE))) {
1961 const auto king_sq = kingSq(~side_to_move_);
1962 const auto file = move.to() > king_sq ? CastleSide::KING_SIDE : CastleSide::QUEEN_SIDE;
1963
1964 if (castling_rights_.getRookFile(~side_to_move_, file) ==
1965 utils::squareFile(move.to())) {
1966 const auto idx = castling_rights_.clearCastlingRight(~side_to_move_, file);
1967 hash_key_ ^= zobrist::castlingIndex(idx);
1968 }
1969 }
1970 }
1971
1972 if (pt == PieceType::KING && castling_rights_.hasCastlingRight(side_to_move_)) {
1973 hash_key_ ^= zobrist::castling(castling_rights_.getHashIndex());
1974
1975 castling_rights_.clearCastlingRight(side_to_move_);
1976
1977 hash_key_ ^= zobrist::castling(castling_rights_.getHashIndex());
1978
1979 }
1980 else if (pt == PieceType::ROOK && utils::ourBackRank(move.from(), side_to_move_)) {
1981 const auto king_sq = kingSq(side_to_move_);
1982 const auto file = move.from() > king_sq ? CastleSide::KING_SIDE : CastleSide::QUEEN_SIDE;
1983
1984 if (castling_rights_.getRookFile(side_to_move_, file) == utils::squareFile(move.from())) {
1985 const auto idx = castling_rights_.clearCastlingRight(side_to_move_, file);
1986 hash_key_ ^= zobrist::castlingIndex(idx);
1987 }
1988 }
1989 else if (pt == PieceType::PAWN) {
1990 half_moves_ = 0;
1991
1992 const auto possible_ep = static_cast<Square>(move.to() ^ 8);
1993 if (std::abs(int(move.to()) - int(move.from())) == 16) {
1994 U64 ep_mask = attacks::pawn(side_to_move_, possible_ep);
1995
1996 if (ep_mask & pieces(PieceType::PAWN, ~side_to_move_)) {
1997 enpassant_sq_ = possible_ep;
1998
1999 hash_key_ ^= zobrist::enpassant(utils::squareFile(enpassant_sq_));
2000 assert(at(enpassant_sq_) == Piece::NONE);
2001 }
2002 }
2003 }
2004
2005 if (move.typeOf() == Move::CASTLING) {
2006 assert(at<PieceType>(move.from()) == PieceType::KING);
2007 assert(at<PieceType>(move.to()) == PieceType::ROOK);
2008
2009 bool king_side = move.to() > move.from();
2010 auto rookTo =
2011 utils::relativeSquare(side_to_move_, king_side ? Square::SQ_F1 : Square::SQ_D1);
2012 auto kingTo =
2013 utils::relativeSquare(side_to_move_, king_side ? Square::SQ_G1 : Square::SQ_C1);
2014
2015 const auto king = at(move.from());
2016 const auto rook = at(move.to());
2017
2018 removePiece(king, move.from());
2019 removePiece(rook, move.to());
2020
2021 assert(king == utils::makePiece(side_to_move_, PieceType::KING));
2022 assert(rook == utils::makePiece(side_to_move_, PieceType::ROOK));
2023
2024 placePiece(king, kingTo);
2025 placePiece(rook, rookTo);
2026
2027 hash_key_ ^= zobrist::piece(king, move.from()) ^ zobrist::piece(king, kingTo);
2028 hash_key_ ^= zobrist::piece(rook, move.to()) ^ zobrist::piece(rook, rookTo);
2029 }
2030 else if (move.typeOf() == Move::PROMOTION) {
2031 const auto piece_pawn = utils::makePiece(side_to_move_, PieceType::PAWN);
2032 const auto piece_prom = utils::makePiece(side_to_move_, move.promotionType());
2033
2034 removePiece(piece_pawn, move.from());
2035 placePiece(piece_prom, move.to());
2036
2037 hash_key_ ^=
2038 zobrist::piece(piece_pawn, move.from()) ^ zobrist::piece(piece_prom, move.to());
2039 }
2040 else {
2041 assert(at(move.from()) != Piece::NONE);
2042 assert(at(move.to()) == Piece::NONE);
2043 const auto piece = at(move.from());
2044
2045 removePiece(piece, move.from());
2046 placePiece(piece, move.to());
2047
2048 hash_key_ ^= zobrist::piece(piece, move.from()) ^ zobrist::piece(piece, move.to());
2049 }
2050
2051 if (move.typeOf() == Move::ENPASSANT) {
2052 assert(at<PieceType>(move.to() ^ 8) == PieceType::PAWN);
2053
2054 const auto piece = utils::makePiece(~side_to_move_, PieceType::PAWN);
2055
2056 removePiece(piece, Square(int(move.to()) ^ 8));
2057
2058 hash_key_ ^= zobrist::piece(piece, Square(int(move.to()) ^ 8));

2381 template <Color c, MoveGenType mt>
2382 void generatePawnMoves(const Board& board, Movelist& moves, Bitboard pin_d, Bitboard pin_hv,
2383 Bitboard checkmask, Bitboard occ_enemy) {
2384 const auto pawns = board.pieces(PieceType::PAWN, c);
2385
2386 constexpr Direction UP = c == Color::WHITE ? Direction::NORTH : Direction::SOUTH;
2387 constexpr Direction DOWN = c == Color::WHITE ? Direction::SOUTH : Direction::NORTH;
2388 constexpr Direction DOWN_LEFT =
2389 c == Color::WHITE ? Direction::SOUTH_WEST : Direction::NORTH_EAST;
2390 constexpr Direction DOWN_RIGHT =
2391 c == Color::WHITE ? Direction::SOUTH_EAST : Direction::NORTH_WEST;
2392
2393 constexpr Bitboard RANK_B_PROMO = c == Color::WHITE
2394 ? attacks::MASK_RANK[static_cast<int>(Rank::RANK_7)]
2395 : attacks::MASK_RANK[static_cast<int>(Rank::RANK_2)];
2396 constexpr Bitboard RANK_PROMO = c == Color::WHITE
2397 ? attacks::MASK_RANK[static_cast<int>(Rank::RANK_8)]
2398 : attacks::MASK_RANK[static_cast<int>(Rank::RANK_1)];
2399 constexpr Bitboard DOUBLE_PUSH_RANK = c == Color::WHITE
2400 ? attacks::MASK_RANK[static_cast<int>(Rank::RANK_3)]
2401 : attacks::MASK_RANK[static_cast<int>(Rank::RANK_6)];
2402
2403 // These pawns can maybe take Left or Right
2404 const Bitboard pawns_lr = pawns & ~pin_hv;
2405
2406 const Bitboard unpinnedpawns_lr = pawns_lr & ~pin_d;
2407 const Bitboard pinnedpawns_lr = pawns_lr & pin_d;
2408
2409 Bitboard l_pawns = (attacks::pawnLeftAttacks<c>(unpinnedpawns_lr)) |
2410 (attacks::pawnLeftAttacks<c>(pinnedpawns_lr) & pin_d);
2411
2412 Bitboard r_pawns = (attacks::pawnRightAttacks<c>(unpinnedpawns_lr)) |
2413 (attacks::pawnRightAttacks<c>(pinnedpawns_lr) & pin_d);
2414
2415 // Prune moves that don't capture a piece and are not on the checkmask.
2416 l_pawns &= occ_enemy & checkmask;
2417 r_pawns &= occ_enemy & checkmask;
2418
2419 // These pawns can walk Forward
2420 const Bitboard pawns_hv = pawns & ~pin_d;
2421
2422 const Bitboard pawns_pinned_hv = pawns_hv & pin_hv;
2423 const Bitboard pawns_unpinned_hv = pawns_hv & ~pin_hv;
2424
2425 // Prune moves that are blocked by a piece
2426 const Bitboard single_push_unpinned = attacks::shift<UP>(pawns_unpinned_hv) & ~board.occ();
2427 const Bitboard single_push_pinned = attacks::shift<UP>(pawns_pinned_hv) & pin_hv & ~board.occ();
2428
2429 // Prune moves that are not on the checkmask.
2430 Bitboard single_push = (single_push_unpinned | single_push_pinned) & checkmask;
2431
2432 Bitboard double_push =
2433 ((attacks::shift<UP>(single_push_unpinned & DOUBLE_PUSH_RANK) & ~board.occ()) |
2434 (attacks::shift<UP>(single_push_pinned & DOUBLE_PUSH_RANK) & ~board.occ())) &
2435 checkmask;
2436
2437 if (pawns & RANK_B_PROMO) {
2438 Bitboard promo_left = l_pawns & RANK_PROMO;
2439 Bitboard promo_right = r_pawns & RANK_PROMO;
2440 Bitboard promo_push = single_push & RANK_PROMO;
2441
2442 // Skip capturing promotions if we are only generating quiet moves.
2443 // Generates at ALL and CAPTURE
2444 while (mt != MoveGenType::QUIET && promo_left) {
2445 const auto index = builtin::poplsb(promo_left);
2446 moves.add(Move::make<Move::PROMOTION>(index + DOWN_RIGHT, index, PieceType::QUEEN));
2447 moves.add(Move::make<Move::PROMOTION>(index + DOWN_RIGHT, index, PieceType::ROOK));
2448 moves.add(Move::make<Move::PROMOTION>(index + DOWN_RIGHT, index, PieceType::BISHOP));
2449 moves.add(Move::make<Move::PROMOTION>(index + DOWN_RIGHT, index, PieceType::KNIGHT));
2450 }
2451
2452 // Skip capturing promotions if we are only generating quiet moves.
2453 // Generates at ALL and CAPTURE
2454 while (mt != MoveGenType::QUIET && promo_right) {
2455 const auto index = builtin::poplsb(promo_right);
2456 moves.add(Move::make<Move::PROMOTION>(index + DOWN_LEFT, index, PieceType::QUEEN));
2457 moves.add(Move::make<Move::PROMOTION>(index + DOWN_LEFT, index, PieceType::ROOK));
2458 moves.add(Move::make<Move::PROMOTION>(index + DOWN_LEFT, index, PieceType::BISHOP));
2459 moves.add(Move::make<Move::PROMOTION>(index + DOWN_LEFT, index, PieceType::KNIGHT));
2460 }
2461
2462 // Skip quiet promotions if we are only generating captures.
2463 // Generates at ALL and QUIET
2464 while (mt != MoveGenType::CAPTURE && promo_push) {
2465 const auto index = builtin::poplsb(promo_push);
2466 moves.add(Move::make<Move::PROMOTION>(index + DOWN, index, PieceType::QUEEN));
2467 moves.add(Move::make<Move::PROMOTION>(index + DOWN, index, PieceType::ROOK));
2468 moves.add(Move::make<Move::PROMOTION>(index + DOWN, index, PieceType::BISHOP));
2469 moves.add(Move::make<Move::PROMOTION>(index + DOWN, index, PieceType::KNIGHT));
2470 }
2471 }
2472
2473 single_push &= ~RANK_PROMO;
2474 l_pawns &= ~RANK_PROMO;
2475 r_pawns &= ~RANK_PROMO;
2476
2477 while (mt != MoveGenType::QUIET && l_pawns) {
2478 const auto index = builtin::poplsb(l_pawns);
2479 moves.add(Move::make<Move::NORMAL>(index + DOWN_RIGHT, index));
2480 }
2481
2482 while (mt != MoveGenType::QUIET && r_pawns) {
2483 const auto index = builtin::poplsb(r_pawns);
2484 moves.add(Move::make<Move::NORMAL>(index + DOWN_LEFT, index));
2485 }
2486
2487 while (mt != MoveGenType::CAPTURE && single_push) {
2488 const auto index = builtin::poplsb(single_push);
2489 moves.add(Move::make<Move::NORMAL>(index + DOWN, index));
2490 }
2491
2492 while (mt != MoveGenType::CAPTURE && double_push) {
2493 const auto index = builtin::poplsb(double_push);
2494 moves.add(Move::make<Move::NORMAL>(index + DOWN + DOWN, index));
2495 }

2497 const Square ep = board.enpassantSq();

2499 const Square epPawn = ep + DOWN;

2501 const Bitboard ep_mask = (1ull << epPawn) | (1ull << ep);
2502
2503 /*
2504 In case the en passant square and the enemy pawn
2505 that just moved are not on the checkmask
2506 en passant is not available.
2507 */
2508 if ((checkmask & ep_mask) == 0) return;
2509
2510 const Square kSQ = board.kingSq(c);
2511 const Bitboard kingMask =
2512 (1ull << kSQ) & attacks::MASK_RANK[static_cast<int>(utils::squareRank(epPawn))];
2513 const Bitboard enemyQueenRook =
2514 board.pieces(PieceType::ROOK, ~c) | board.pieces(PieceType::QUEEN, ~c);
2515
2516 const bool isPossiblePin = kingMask && enemyQueenRook;

2518
2519 // For one en passant square two pawns could potentially take there.
2520
2521 while (epBB) {
2522 const Square from = builtin::poplsb(epBB);
2523 const Square to = ep;
2524
2525 /*
2526 If the pawn is pinned but the en passant square is not on the
2527 pin mask then the move is illegal.
2528 */
2529 if ((1ULL << from) & pin_d && !(pin_d & (1ull << ep))) continue;

2531 const Bitboard connectingPawns = (1ull << epPawn) | (1ull << from);
2532
2533 /*
2534 7k/4p3/8/2KP3r/8/8/8/8 b - - 0 1

2536 However, we cannot take en passant because that would put our king
2537 in check. For this scenario we check if there's an enemy rook/queen
2538 that would give check if the two pawns were removed.
2539 If that's the case then the move is illegal and we can break immediately.
2540 */
2541 if (isPossiblePin &&
2542 (attacks::rook(kSQ, board.occ() & ~connectingPawns) & enemyQueenRook) != 0)
2543 break;
2544
2545 moves.add(Move::make<Move::ENPASSANT>(from, to));
2546 }
2547 }

2689 Bitboard _occ_all = _occ_us | _occ_enemy;
2690 Bitboard _enemy_emptyBB = ~_occ_us;
2691
2692 Bitboard _checkMask = checkMask<c>(board, king_sq, _doubleCheck);
2693 Bitboard _pinHV = pinMaskRooks<c>(board, king_sq, _occ_enemy, _occ_us);
2694 Bitboard _pinD = pinMaskBishops<c>(board, king_sq, _occ_enemy, _occ_us);
2695
2696 assert(_doubleCheck <= 2);
2697
2698 // Moves have to be on the checkmask
2699 Bitboard movable_square;
2700
2701 // Slider, Knights and King moves can only go to enemy or empty squares.
2702 if (mt == MoveGenType::ALL)
2703 movable_square = _enemy_emptyBB;
2704 else if (mt == MoveGenType::CAPTURE)
2705 movable_square = _occ_enemy;
2706 else // QUIET moves
2707 movable_square = ~_occ_all;
2708
2709 if (pieces & PieceGenType::KING) {
2710 Bitboard _seen = seenSquares<~c>(board, _enemy_emptyBB);
2711
2712 whileBitboardAdd(movelist, 1ull << king_sq,
2713 [&](Square sq) { return generateKingMoves(sq, _seen, movable_square); });
2714
2715 if (utils::squareRank(king_sq) == (c == Color::WHITE ? Rank::RANK_1 : Rank::RANK_8) &&
2716 (board.castlingRights().hasCastlingRight(c) &&
2717 _checkMask == constants::DEFAULT_CHECKMASK)) {
2718 Bitboard moves_bb = generateCastleMoves<c, mt>(board, king_sq, _seen, _pinHV);
2719
2720 while (moves_bb) {
2721 Square to = builtin::poplsb(moves_bb);
2722 movelist.add(Move::make<Move::CASTLING>(king_sq, to));
2723 }
2724 }
2725 }
2726
2727 movable_square &= _checkMask;
2728
2729 // Early return for double check as described earlier
2730 if (_doubleCheck == 2) return;
2731
2732 // Add the moves to the movelist.
2733 if (pieces & PieceGenType::PAWN) {
2734 generatePawnMoves<c, mt>(board, movelist, _pinD, _pinHV, _checkMask, _occ_enemy);
2735 }
2736
2737 if (pieces & PieceGenType::KNIGHT) {
2738 // Prune knights that are pinned since these cannot move.
2739 Bitboard knights_mask = board.pieces(PieceType::KNIGHT, c) & ~(_pinD | _pinHV);
2740
2741 whileBitboardAdd(movelist, knights_mask,
2742 [&](Square sq) { return generateKnightMoves(sq) & movable_square; });
2743 }
2744
2745 if (pieces & PieceGenType::BISHOP) {
2746 // Prune horizontally pinned bishops
2747 Bitboard bishops_mask = board.pieces(PieceType::BISHOP, c) & ~_pinHV;
2748
2749 whileBitboardAdd(movelist, bishops_mask, [&](Square sq) {
2750 return generateBishopMoves(sq, _pinD, _occ_all) & movable_square;
2751 });
2752 }
2753
2754 if (pieces & PieceGenType::ROOK) {
2755 // Prune diagonally pinned rooks
2756 Bitboard rooks_mask = board.pieces(PieceType::ROOK, c) & ~_pinD;
2757
2758 whileBitboardAdd(movelist, rooks_mask, [&](Square sq) {
2759 return generateRookMoves(sq, _pinHV, _occ_all) & movable_square;
2760 });
2761 }
2762
2763 if (pieces & PieceGenType::QUEEN) {
2764 // Prune double pinned queens
2765 Bitboard queens_mask = board.pieces(PieceType::QUEEN, c) & ~(_pinD & _pinHV);
2766
2767 whileBitboardAdd(movelist, queens_mask, [&](Square sq) {
2768 return generateQueenMoves(sq, _pinD, _pinHV, _occ_all) & movable_square;
2769 });
2770 }
